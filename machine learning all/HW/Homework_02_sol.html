% K-means

% Note: this script will probably take 5-10 min in Octave, with nothing
% printing to the screen and no figures appearing until the entire script 
% is done.

clear all; close all; clc;


% Load small image
im = imread('bird.tiff');
% im = imread('handwritten.jpg');
A = double(im);

figure(1),  clf
imshow(im)

dim = size(A,1); % number of pixels in picture's length/width
k = 16; % number of colors to represent

%%
% Initialize means to randomly-selected colors in the original photo.
means = zeros(k, 3);
means_new = zeros(k, 3);
rand_x = ceil(dim*rand(k, 1));
rand_y = ceil(dim*rand(k, 1));
for i = 1:k
    means(i,:) = A(rand_x(i), rand_y(i), :);
end

clc

A16 = reshape(A,[dim*dim,3]);
B = zeros(dim*dim,1);

for j = 1:100
    for i = 1:dim*dim
        [~, I] = min(sum((means - repmat(A16(i,:),[k,1])).^2,2));
        B(i) = I;
    end

    %
    for i = 1:k
        means_new(i,:) = mean(A16((B == i),:),1);
    end
    
     % Convergence test. Display by how much the means values are changing
    d = sum(sqrt(sum((means_new - means).^2, 2)));
    if d < 1
        break
    end
    
    means = means_new;
    
end

for i = 1:dim*dim
    A16(i,:) = round(means(B(i),:));
end

figure(2),  clf
imshow(uint8(reshape(A16,[dim,dim,3])))

% Save image
% imwrite(uint8(round(large_image)), 'bird_kmeans.jpg');

%%
figure(3),  clf
for i = 1:k
   col = (1/255).*means(i,:);
   rectangle('Position', [i, 0, 1, 1], 'FaceColor', col, 'EdgeColor', col);
end

Problem 2¶

Distance from the line to arbitrary vector x

$$h = \frac{g(x)}{\lVert \omega \rVert}$$$$\begin{align*} f(x_1, x_2) &= 2x_1 + 3x_2 + 1 = 0\\ f(\vec x) &= \begin{bmatrix}2& 3\end{bmatrix} \begin{bmatrix}x_1\\ x_2\end{bmatrix} + 1 = \omega^Tx + 1 = 0 \\ \end{align*}$$$$\begin{align*} h_a &= \frac{g(x_a)}{\lVert \omega \rVert} = \frac{(4+3+1)}{\sqrt{13}} \frac{8}{\sqrt{13}}\\ h_b &= \frac{g(x_b)}{\lVert \omega \rVert} = \frac{2-3+1}{\sqrt{13}} = 0 \end{align*}$$

Problem 3¶

If $X$ is on the linf of $\omega^TX + \omega_0 = 0$, the objective function can be interpreted as the shortest distance $r^{*}$ from origin to the line. We geometrically (or intuitively) know that distance $r$ will be minimal when vector $X$ is orthogonal to the line


$$r^{*} = -\frac{\omega_0}{\lVert \omega \rVert} \text{and} \vec X^* = -\frac{\omega_0}{\lVert \omega \rVert}\frac{\vec \omega}{\lVert \omega \rVert} = \frac{\omega_0}{\omega^T\omega}\vec \omega$$

Problem 4¶

$$\begin{align*} \sigma(z) &= \frac{1}{1+e^{-x}}\\ \frac{d\sigma(z) }{dx} &= \frac{(1 + e^{-x})0 - 1(-e^{-x})}{(1 + e^{-1})^2} &= \frac{e^{-x}}{(1+e^{-x})^2}\\ \end{align*}$$

note $$\begin{align*} 1 - \sigma(z) &= \frac{1+e^{-x}-1}{1+e^{-x}} &= \frac{e^{-x}}{1+e^{-x}} \end{align*}$$

hence

$$\frac{d\sigma(z)}{dx} = (1-\sigma(z)) \sigma(z) $$

Problem 6¶

$X, Y = aX + b$, We know that two r.v.s are linear

$$\rho(X,Y) = 1 \quad a > 0$$$$\rho(X,Y) = -1 \quad a < 0$$

$\rho$ indicates how linear

$$\begin{align*} \rho = \frac{cov(X, Y)}{\sqrt{VAR(X)}\sqrt{VAR(Y)}} = \frac{E[(X-\bar X)(Y-\bar Y)]}{\sqrt{E[(X-\bar X)^2]}\sqrt{E[(Y-\bar Y)^2]}} \end{align*}$$$$ \begin{align*} E[(X-\bar X)(aX-a\bar X)] & = a E[(X-\bar X)^2]\\ E[(Y - \bar Y)] &= E[a^2(X- \bar X)^2] = a^2E[(X-\bar X)^2] \end{align*} $$
